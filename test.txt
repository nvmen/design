import { fixture, html, expect } from 'vitest';
import '../my-element.js';

describe('my-element', () => {
  it('displays the default message', async () => {
    const el = await fixture(html`<my-element></my-element>`);
    expect(el.shadowRoot.textContent.trim()).to.equal('Hello, World!');
  });

  it('displays a custom message', async () => {
    const el = await fixture(html`<my-element message="Test"></my-element>`);
    expect(el.shadowRoot.textContent.trim()).to.equal('Test');
  });
});
